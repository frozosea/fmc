    function getFormattedDate(date1){
        var TRIM_PATTERN = /(^\s*)|(\s*$)/g;

        var date = date1.replace(TRIM_PATTERN,"");
        //영문 홈페이지일때 날짜표기변환 2011.09.09 jjh 추가



        if(date!="" ){
            return date.substring(2,4)+"/"+date.substring(4,6)+"/"+date.substring(6,8);
        }else{
            return date;
        }



    }
