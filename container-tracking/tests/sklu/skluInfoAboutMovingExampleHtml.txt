                            dataType: 'JSON',

                            success: function (result) {
                                if (result.STATUS == "N") {
                                    alert(result.MSG);
                                    return false;

                                    location.href = "../BLDetail?blno=" + $("#_laySearchBox").val();

                $("#_laySearchBox").autocomplete({
                    source: function (request, response) {
                        $.ajax({
                            url: "/Home/GetNoList",
                            type: "POST",
                            dataType: "json",
                            data: {Prefix: request.term.toUpperCase()},
                            success: function (data) {
                                response($.map(data, function (item) {
                                    return {
                                        label: item.Text,
                                        value: item.Text
                                    }
                                }))
                            },
                            error: function (response) {
                                alert(response.responseText);
                            }
                        });
                    },
                    select: function (e, i) {
                        if (i.item.value.indexOf("(") > -1)

                $("body").addClass("hide-sidebar");
            } else {
                $("body").removeClass("hide-sidebar");
            }
        });

        var setCookie = function (name, value, exp) {
            var date = new Date();
            date.setTime(date.getTime() + exp * 24 * 60 * 60 * 1000);
            document.cookie = name + '=' + value + ';expires=' + date.toUTCString() + ';path=/';
        };

        function hideMenu() {
            //쿠키 설정
            if ($("body").hasClass("hide-sidebar") == false) {
                setCookie("hideMenu_Sinokor_eService", "true", 1);
            } else {
                setCookie("hideMenu_Sinokor_eService", "false", 1);
            }
        }

        function IsSetCookie(name, val) {
            var nameOfCookie = name + "=" + val;

            if (document.cookie.indexOf(nameOfCookie) == -1) {
                return false;
            } else {
                return true;
            }
        }

    if (window.performance) {
        var navEntries = window.performance.getEntriesByType('navigation');
        if (navEntries.length > 0 && navEntries[0].type === 'back_forward') {
            $("#isHistoryBackOrFowerd").val("true");
            //console.log('As per API lv2, this page is load from back/forward');
        } else if (window.performance.navigation
            && window.performance.navigation.type == window.performance.navigation.TYPE_BACK_FORWARD) {
            $("#isHistoryBackOrFowerd").val("true");
            //console.log('As per API lv1, this page is load from back/forward');
        } else {
            $("#isHistoryBackOrFowerd").val("false");
            //console.log('This is normal page load');
        }
    } else {
        $("#isHistoryBackOrFowerd").val("false");
        //console.log("Unfortunately, your browser doesn't support this API");
    }

    $(document).on('submit', 'form', function () {
        var buttons = $(this).find('[type="submit"]');
        if ($(this).valid()) {
            buttons.each(function (btn) {
                $(buttons[btn]).prop('disabled', true);
            });
        } else {
            buttons.each(function (btn) {

            });

    }).ready(function () {
        if (IsSetCookie("hideIE_Sinokor", "true")) {
            $("#ieFooter").remove();
            $("#normalFooter").show();
        } else {
            var agent = navigator.userAgent.toLowerCase();
            if ((navigator.appName == 'Netscape' && navigator.userAgent.search('Trident') != -1) || (agent.indexOf("msie") != -1)) {
                $("#ieFooter").show();
                $("#normalFooter").remove();
            } else {
                $("#ieFooter").remove();
                $("#normalFooter").show();
            }
        }

        if ("False" == "True" && "RU" == "KR") {
            if (IsSetCookie("hideLoadingConfirm_Sinokor", "true") == false) {
                $.ajax({
                    url: "/Booking/GetLoadingConfirmCount",
                    type: "POST",
                    dataType: "json",
                    data: {},
                    success: function (result) {
                        if (Number(result) > 0) {
                            $("#lblLoadingConfirmCnt").html("");
                            $("#lblLoadingConfirmCnt").append(result);
                            setInterval(toggleLoadingConfirm, 60000);
                        }
                    }
                });
            }
        }
    });

    $('input[type="text"]').keydown(function (event) {
        if (event.keyCode == 13) {
            event.preventDefault();
            return false;
        }
    });

    $.ajax({
        url: "/Notice/GetRecentNotice",
        type: "POST",
        dataType: "json",
        data: {},
        success: function (result) {
            if (result.length > 0) {
                $("#carouselBody").text("");
                var html = "";
                $.each(result, function (i, data) {
                    if (i == 0)
                        html += "<div class='item active'> ";
                    else
                        html += "<div class='item'> ";
                    html += "<img src='../../Content/Image/CarouselImg.png' />"
                        + "<div class='carousel-caption d-none d-md-block' style='background-color: transparent; padding-top: 25px;'>"
                        + "<a style='color: white;' href=\"javascript: viewNoticeDetail('" + data.NUM + "'); \"><h5>" + data.TITLE + "</h5></a>"
                        + "</div>"
                        + "</div>";
                });
                $("#carouselBody").append(html);
                $("#noticeDiv").addClass("carousel slide");
            }
        },
        error: function (response) {
            alert(response.responseText);
        }
    });

    function toggleLoadingConfirm() {
        var obj = $("#divLoadingConfirm");
        if (obj.css("display") == "none") {
            var height = obj.css({
                display: "block"
            }).height();

            obj.css({
                overflow: "hidden",
                marginBottom: height,
                height: 0
            }).animate({
                marginBottom: 0,
                height: height
            }, 500, function () {
                $(this).css({
                    overflow: "",
                    height: "",
                    marginTop: ""
                });
                obj.show();
                $("#btnLoadingConfirm").show();
            });
        } else {
            var height = obj.css({
                display: "block"
            }).height();
            $("#btnLoadingConfirm").hide();
            obj.css({
                overflow: "hidden",
                //marginTop: height,
                marginBottom: 0,
                height: height
            }).animate({
                marginBottom: height,
                height: 0
            }, 500, function () {
                $(this).css({
                    overflow: "",
                    height: "",
                    marginTop: ""
                });
                obj.hide();
            });
        }
    }
